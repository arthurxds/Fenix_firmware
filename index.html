#include #include #include #include #include #include #include #include #include // Pinos #define PIN_SDA 21 #define PIN_SCL 22 #define PIN_SPI_MISO 19 #define PIN_SPI_MOSI 23 #define PIN_SPI_SCK 18 #define PIN_LORA_NSS 5 #define PIN_LORA_RST 32 #define PIN_LORA_DIO0 33 #define PIN_SD_CS 15 #define PIN_GPS_RX 16 #define PIN_GPS_TX 17 #define PIN_BUZZER 25 #define PIN_SQUIB 26 #define PIN_RBF 34 // Constantes #define LOOP_MS 20 #define TELEMETRY_MS 500 #define SD_SYNC_MS 1000 enum FlightState : uint8_t { IDLE, VOANDO, APOGEU, DESCIDA, SOLO }; // Variáveis Globais FlightState currentState = IDLE; float padPressure = 101325.0, padAlt = 0.0, pressure = 0.0, temp = 0.0; float altAGL = 0.0, altPrev = 0.0, vz = 0.0, maxAlt = 0.0; float ax = 0.0, ay = 0.0, az = 0.0; double gpsLat = 0.0, gpsLon = 0.0; float gpsAlt = 0.0; uint8_t gpsSats = 0; bool ms5611_ok = false, adxl_ok = false, lora_ok = false, sd_ok = false; bool gpsValid = false, rbfRemoved = false, squibFired = false; int launchCount = 0, apogeeCount = 0; unsigned long lastLoop = 0, lastTele = 0, lastSync = 0, apogeeTimer = 0, landingTimer = 0; String dataString = ""; // Objetos MS5611 baro; Adafruit_ADXL375 accel(0x53); RH_RF95 lora(PIN_LORA_NSS, PIN_LORA_DIO0); TinyGPSPlus gps; HardwareSerial gpsSerial(2); File logFile; // Protótipos void readSensors(); void updateFSM(); void processData(); void fireSquib(); void checkRBF(); void setup() { Serial.begin(115200); pinMode(PIN_BUZZER, OUTPUT); pinMode(PIN_SQUIB, OUTPUT); pinMode(PIN_RBF, INPUT); digitalWrite(PIN_BUZZER, LOW); digitalWrite(PIN_SQUIB, LOW); Wire.begin(PIN_SDA, PIN_SCL); SPI.begin(PIN_SPI_SCK, PIN_SPI_MISO, PIN_SPI_MOSI); gpsSerial.begin(9600, SERIAL_8N1, PIN_GPS_RX, PIN_GPS_TX); // Inicia SD if (SD.begin(PIN_SD_CS)) { logFile = SD.open("/voo_log.csv", FILE_WRITE); if (logFile) { logFile.println("ms,estado,alt,vz,pressao,temp,ax,ay,az,lat,lon,sats,rbf,squib"); sd_ok = true; } } // Inicia Barômetro if (baro.begin()) { baro.setOversampling(OSR_ULTRA_HIGH); ms5611_ok = true; } // Inicia Acelerômetro if (accel.begin()) { accel.setDataRate(ADXL343_DATARATE_100_HZ); adxl_ok = true; } // Inicia LoRa pinMode(PIN_LORA_RST, OUTPUT); digitalWrite(PIN_LORA_RST, HIGH); if (lora.init()) { lora.setFrequency(915.0); lora.setTxPower(20, false); lora_ok = true; } // Calibração rápida do Pad delay(2000); if (ms5611_ok && baro.read() == 0) padPressure = baro.getPressure(); Serial.println("Fenix UFMG - Sistema Armado e Calibrado!"); } void loop() { unsigned long now = millis(); if (now - lastLoop < LOOP_MS) return; lastLoop = now; while (gpsSerial.available()) gps.encode(gpsSerial.read()); if (gps.location.isValid()) { gpsLat = gps.location.lat(); gpsLon = gps.location.lng(); gpsAlt = gps.altitude.meters(); gpsSats = gps.satellites.value(); gpsValid = true; } readSensors(); updateFSM(); processData(); // Salva no SD e envia LoRa } void readSensors() { checkRBF(); if (ms5611_ok && baro.read() == 0) { pressure = baro.getPressure(); temp = baro.getTemperature(); altPrev = altAGL; // Fórmula AGL simplificada altAGL = 44330.0 * (1.0 - pow(pressure / padPressure, 0.1903)); vz = (altAGL - altPrev) / (LOOP_MS / 1000.0); if (altAGL > maxAlt) maxAlt = altAGL; } if (adxl_ok) { sensors_event_t event; accel.getEvent(&event); ax = event.acceleration.x / 9.81; ay = event.acceleration.y / 9.81; az = event.acceleration.z / 9.81; } } void updateFSM() { switch (currentState) { case IDLE: if (altAGL > 10.0) launchCount++; else launchCount = 0; if (launchCount > 3) { currentState = VOANDO; Serial.println("LANCAMENTO!"); } break; case VOANDO: if (vz < 0.0) apogeeCount++; else apogeeCount = 0; if (apogeeCount > 3) { currentState = APOGEU; apogeeTimer = millis(); Serial.println("APOGEU DETECTADO!"); } break; case APOGEU: // Espera 1 segundo de forma NÃO-bloqueante para acionar o squib backup if (millis() - apogeeTimer >= 1000 && !squibFired) { fireSquib(); currentState = DESCIDA; } break; case DESCIDA: if (abs(vz) < 0.5) { if (landingTimer == 0) landingTimer = millis(); if (millis() - landingTimer > 3000) currentState = SOLO; } else { landingTimer = 0; } break; case SOLO: if (millis() % 3000 < 100) digitalWrite(PIN_BUZZER, HIGH); // Bipa a cada 3s else digitalWrite(PIN_BUZZER, LOW); break; } } void fireSquib() { if (!rbfRemoved || altAGL < 50.0 || squibFired) return; digitalWrite(PIN_SQUIB, HIGH); delay(500); // 500ms de pulso é seguro aqui pois já estamos em descida digitalWrite(PIN_SQUIB, LOW); squibFired = true; Serial.println("SQUIB DISPARADO!"); } void checkRBF() { rbfRemoved = (digitalRead(PIN_RBF) == HIGH); } void processData() { unsigned long now = millis(); // Monta a string CSV uma vez só dataString = String(now) + "," + String(currentState) + "," + String(altAGL,1) + "," + String(vz,1) + "," + String(pressure,0) + "," + String(temp,1) + "," + String(ax,2) + "," + String(ay,2) + "," + String(az,2) + "," + String(gpsLat,6) + "," + String(gpsLon,6) + "," + String(gpsSats) + "," + String(rbfRemoved) + "," + String(squibFired); // SD Card (50Hz) if (sd_ok) { logFile.println(dataString); if (now - lastSync > SD_SYNC_MS) { logFile.flush(); lastSync = now; } } // Telemetria LoRa (2Hz) if (lora_ok && (now - lastTele > TELEMETRY_MS)) { lora.send((uint8_t*)dataString.c_str(), dataString.length()); lora.waitPacketSent(); lastTele = now; } }
